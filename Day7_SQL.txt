--------------------- Text Functions

use batch

Select * from Salesorder

---- LEN -- CHARACTER COUNT

SELECT CITY , LEN(CITY) AS LEN FROM Salesorder

SELECT CITY , [CUSTOMER ID] , LEFT([CUSTOMER ID],2)  AS CHART FROM Salesorder

SELECT CITY , [CUSTOMER ID] , LEFT([CUSTOMER ID],2)  AS CHART , RIGHT([CUSTOMER ID],5) AS CHARTR FROM Salesorder

 ----------Upper , Lower , Substring 

 SELECT CITY , LOWER(CITY) AS LCITY ,UPPER(CITY) AS UCITY , SUBSTRING(CITY,2,5) AS SCITY FROM Salesorder


 ----------------- MATH FUNCTIONS(AGG FUNCTIONS)

 ------ SUM , COUNT , MIN , MAX, AVG 

 SELECT SUM(SALES) AS TOTAL , COUNT(SALES) AS COUNT FROM Salesorder

 SELECT MIN(SALES) AS MTOTAL , MAX(SALES) AS MTOTL FROM Salesorder

 SELECT AVG(SALES) AS ATOTAL , AVG(Quantity) AS ATOTL FROM Salesorder


 ------------- ROUND 

  SELECT ROUND(SUM(SALES),2) AS RTOTAL , COUNT(SALES) AS COUNT FROM Salesorder
