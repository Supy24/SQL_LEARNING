----1- Comparion Operators - > , < , >= , <= , <>  or =! , Between... and 

----WAQ to fetch the country, city , state  where sales>500

SELECT COUNTRY , STATE, CITY , SALES  FROM Salesorder WHERE SALES>500

----WAQ to fetch COUNTRY , STATE, CITY , SALES where sales <= 1000

SELECT COUNTRY , STATE, CITY , SALES  FROM Salesorder WHERE SALES<=500

----WAQ to fetch COUNTRY , STATE, CITY , SALES where quantity = 5

SELECT COUNTRY , STATE, CITY , SALES , Quantity FROM Salesorder WHERE Quantity =5

----WAQ to fetch COUNTRY , STATE, CITY , SALES where  not  quantity = 5

SELECT COUNTRY , STATE, CITY , SALES , Quantity FROM Salesorder WHERE Quantity  <> 5


----WAQ to fetch COUNTRY , STATE, CITY , SALES where   quantity   5 to 8


SELECT COUNTRY , STATE, CITY , SALES , Quantity FROM Salesorder WHERE Quantity  between 5 and 8
